# Build the Docker image for the current folder 
# and tag it with `dockerized-react`

docker build . -t budemiv5

# Check the image was created
docker images | grep budemiv5


docker tag budemiv5 kadirgokhann/budemiv5

docker push kadirgokhann/budemiv5

docker pull kadirgokhann/budemiv5


# Run the image in detached mode 
# and map port 3000 inside the container with 3000 on current host
docker run -p 80:3000 -d kadirgokhann/budemiv5
