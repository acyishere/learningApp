# Build the Docker image for the current folder 
# and tag it with `dockerized-react`

docker tag budemiv7 kadirgokhann/budemiv7

docker push kadirgokhann/budemiv7


from instance:
docker login --username=kadirgokhann
docker pull kadirgokhann/budemiv7


# Run the image in detached mode 
# and map port 3000 inside the container with 80 on current host
docker run -p 80:3000 -d kadirgokhann/budemiv7
